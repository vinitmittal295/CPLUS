Q find duplicate

 #include <bits/stdc++.h>
using namespace std;
int main(){
    int n;
    cout<<"enter the size";
    cin>>n;
    int arr[n];
    for(int i=0; i<n; i++){
        cin>>arr[i];
    }
    int k=0;
    for(int i=0; i<n; i++){
        if(arr[i]!=arr[k]){
            arr[k]=arr[i];
                 k++;
        }
    }
    for(int i=0; i<=k; i++){	
        cout<<arr[i]<<" ";
    }
    cout<<"\n";
}
